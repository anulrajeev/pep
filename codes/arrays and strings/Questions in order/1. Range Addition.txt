#include <bits/stdc++.h>
using namespace std;

int main()
{
  	int n,q;
  	cin>>n>>q;
  	int A[n]={0};
  	int start, end, inc;
  	for(int i=0;i<q;i++)
  	{
  		cin>>start>>end>>inc;
  		A[start]+=inc;
  		if(end+1<n)
  			A[end+1]+=-inc;
  	}
  	cout<<A[0]<<' ';
  	for(int i=1;i<n;i++)
  	{
  		A[i]+=A[i-1];
  		cout<<A[i]<<' ';
  	}

    return 0;

}
