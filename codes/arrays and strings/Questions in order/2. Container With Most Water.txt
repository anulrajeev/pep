vi convert(int A[], int n)
{
	vi res;
	int beg,mid,end;
	for(mid=0;mid<n;mid++)
		if(A[mid]>=0)
			break;
	end=mid;
	beg=mid-1;
	while(beg>=0 || end<n)
	{
		if(beg>=0 && end<n)
		{
			if(abs(A[beg])<abs(A[end]))
			{
				res.push_back(A[beg]*A[beg]);
				beg--;
			}
			else
			{
				res.push_back(A[end]*A[end]);
				end++;
			}
		}
		else if(end<n)
			{
				res.push_back(A[end]*A[end]);
				end++;
			}
		else 
			{
				res.push_back(A[beg]*A[beg]);
				beg--;
			}

	}
	return res;
}

void _main()
{
  	int n;
  	cin>>n;
  	int A[n];
  	for(int i=0;i<n;i++)
	  	cin>>A[i];
	vi v = convert(A,n);
	for(int i=0;i<n;i++)
		cout<<v[i]<<' ';
}
