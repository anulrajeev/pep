#include <bits/stdc++.h>
using namespace std;
typedef vector<int> vi;
typedef vector<vi> vvi;

int main()
{
  	int m,n;
  	cin>>n;
  	vvi grid(n,vi(n));
  	for(int i=0;i<n;i++)
  		for(int j=0;j<n;j++)
  			cin>>grid[i][j];

  	for(int i=0;i<n;i++)
  		for(int j=i+1;j<n;j++)
  			swap(grid[i][j], grid[j][i]);

  	for(int i=0;i<n;i++)
  		reverse(grid[i].begin(), grid[i].end());

  	for(int i=0;i<n;i++)
  	{
  		for(int j=0;j<n;j++)
  			cout<<grid[i][j]<<' ';
  		cout<<'
';
  	}
  	return 0;
}
