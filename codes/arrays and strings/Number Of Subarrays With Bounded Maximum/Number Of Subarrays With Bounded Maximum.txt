#include <bits/stdc++.h>
using namespace std;

int main()
{
  	int n;
  	cin>>n;
  	int A[n];
  	for(int i=0;i<n;i++)
  		cin>>A[i];
  	int low,high;
  	cin>>low>>high;
  	int start=0, i, N=n, prev=0, res=0;
  	for(i=0;i<N;i++)

  		if(A[i]>=low && A[i]<=high)		//normal elements
  		{
  			res+=i-start+1;
            prev=i-start+1;
  		}
  		else if(A[i]<low)				//ok elements
  		{
  			res+=prev;
  		}
  		else 						//blocking elements
  		{
  			start=i+1;
            prev=0;
  		}
  	cout<<res;
    return 0;
}
