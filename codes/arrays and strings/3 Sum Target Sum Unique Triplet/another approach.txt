#include<bits/stdc++.h>
using namespace std;

typedef pair<int,int> pii;
typedef vector<int> vi;
typedef vector<vi> vvi;


vector<pii> twoSum(int A[], int index, int n, int sum)
{
    vector<pii> res;
    int i=index, j=n-1;
    
    while(i<j)
    {
        if(i>index && A[i]==A[i-1])
            i++;
        else if(A[i]+A[j]==sum)
            res.push_back({A[i++], A[j--]});
        else if(A[i]+A[j]<sum)
            i++;
        else 
            j--;
    }
    return res;
}

void threeSum(int A[], int n, int target)
{
    sort(A,A+n);
    vvi res;
    for(int i=0;i<n;i++)
    {
        if(i>0 && A[i]==A[i-1])
            continue;
            
        vector<pii> v = twoSum(A,i+1,n,target-A[i]);
        
        for(auto p : v)
            res.push_back({A[i], p.first, p.second});
    }
    
    for(auto item : res)
        cout<<item[0]<<' '<<item[1]<<' '<<item[2]<<'\n';
}

int main()
{
    int n;
    cin>>n;
    int A[n];
    for(int i=0;i<n;i++)
        cin>>A[i];
    int target;
    cin>>target;
    threeSum(A,n,target);
    return 0;
}
