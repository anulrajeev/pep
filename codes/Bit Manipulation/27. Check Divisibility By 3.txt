import java.io.*;
import java.util.*;

public class Main {

  public static void main(String[] args){
    Scanner scn = new Scanner(System.in);
    String s= scn.nextLine();

    //write your code here
    int odd=0, even=0;
  	for(int i=s.length()-1;i>=0;i--)
  		if(s.charAt(i)=='1')
  			if((i&1)==0)
  				odd++;
  			else
  				even++;
  	System.out.println(((odd-even)%3==0)?"true":"false");
  }

}
