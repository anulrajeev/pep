void solution(int n)
{
	bool flag=false;
	int rev=0, j=0;
	for(int i=31;i>=0;i--)
	{
		int mask=(1<<i);
		if(flag)
		{
			if(n&mask)
			{
				cout<<1;
				rev|=(1<<j);
			}
			else
				cout<<0;

			j++;
		}
		else
		{
			if(n&mask)
			{
				flag=true;
				cout<<1;
				rev=(1<<j++);
			}
		}
	}
	cout<<'
';
	cout<<rev;
}
