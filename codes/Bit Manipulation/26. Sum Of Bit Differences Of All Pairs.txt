ll f(ll A[], ll n)
{
	ll count = 0;

	for(ll j=0;j<32;j++)
	{
		ll set=0, not_set=0;
		for(ll i=0;i<n;i++)
			if(A[i]&(1<<j))
				set++;
			else
				not_set++;
		count+=set*not_set*2;
	}
	return count;

}
