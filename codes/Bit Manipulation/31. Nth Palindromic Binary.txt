int getRev(int n)
{
	int rev=0;
	while(n)
	{
		rev<<=1;
		rev|=(n&1);
		n>>=1;

	}
	return rev;
}

int solution(int n)
{
	int len=1, count=1;
	while(count<n)
	{
		len++;
		int elementsForthisLen = (1<<(len-1)/2);
		count+=elementsForthisLen;
	}

	count-=(1<<(len-1)/2);
	int offset=(n-count-1);

	int ans=(1<<(len-1));
	ans|=(offset<<len/2);
	ans|=getRev(ans>>(len/2));
	return ans;
}
