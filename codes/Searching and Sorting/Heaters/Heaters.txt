int getRadius(vi &house, vi &heater)
{
	int res=0;
	int m=house.size(), n=heater.size();
	sort(heater.begin(), heater.end());
	for(auto h : house)
	{
		int r = lower_bound(heater.begin(), heater.end(),h)-heater.begin();
		int l=r-1;
		int left_dist = (l>=0?h-heater[l]:INT_MAX);
		int right_dist= (r==n?INT_MAX:heater[r]-h);
		int dist=min(left_dist,right_dist);
		res=max(res,dist);
		// cout<<h<<' '<<left_dist<<' '<<right_dist<<'\n';
	}
	return res;
}

  	vi heater(n);
  	for(int j=0;j<n;j++)
  		cin>>heater[j];
  	cout<<getRadius(house,heater);
}
