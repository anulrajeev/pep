#include<bits/stdc++.h>
using namespace std;

int C(int n, int r)
{
    int res=1;
    for(int i=0;i<r;i++)
    {
        res=res*(n-i);
        res=res/(i+1);
    }
    return res;
}

int getCount(int A[], int n)
{
    int count=0;
    unordered_map<int,int> m;
    for(int i=0;i<n;i++)
        m[A[i]]++;
    for(auto p : m)
        if(p.second>1)
            count+=C(p.second,2);
    return count;
}

int main()
{
    int n;
    cin>>n;
    int A[n];
    for(int i=0;i<n;i++)
        cin>>A[i];
    cout<<getCount(A,n);
    return 0;
}
