#include<bits/stdc++.h>
using namespace std;
typedef vector<int> vi;

vi pancakeSort(vi & A) {
        
        vector<int> res;
        for(int size=A.size();size>0;size--)
        {
            int index=max_element(A.begin(),A.begin()+size)-A.begin();
            if(index==size-1)
                continue;
            res.push_back(index+1);
            reverse(A.begin(),A.begin()+index+1);
            res.push_back(size);
            reverse(A.begin(),A.begin()+size);
        }
        return res;        
    }
int main()
{
    int n;
    cin>>n;
    vi v(n);
    for(int i=0;i<n;i++)
        cin>>v[i];
    vi res= pancakeSort(v);
    cout<<'[';
    if(res.size())
    {
    for(int i=0;i<res.size()-1;i++)
        cout<<res[i]<<", ";
        cout<<res[res.size()-1];
    }
    cout<<']';
    return 0;
}
