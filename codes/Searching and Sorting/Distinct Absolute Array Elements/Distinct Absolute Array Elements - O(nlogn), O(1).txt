#include<bits/stdc++.h>
using namespace std;

int getCount(int A[], int n)
{
   int count=0;
   for(int i=0;i<n;i++)
        if(i==0 || A[i]!=A[i-1])
            if(A[i]>=0)
                count++;
            else if(!binary_search(A,A+n,-A[i]))
                count++;
                
    return count;
}

int main()
{
    int n;
    cin>>n;
    int A[n];
    for(int i=0;i<n;i++)
        cin>>A[i];
    cout<<getCount(A,n);
    return 0;
}
