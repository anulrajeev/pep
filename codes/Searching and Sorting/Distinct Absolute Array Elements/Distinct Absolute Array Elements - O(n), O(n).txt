#include<bits/stdc++.h>
using namespace std;

int getCount(int A[], int n)
{
    unordered_set<int> s;
    for(int i=0;i<n;i++)
        s.insert(abs(A[i]));
    return s.size();
}

int main()
{
    int n;
    cin>>n;
    int A[n];
    for(int i=0;i<n;i++)
        cin>>A[i];
    cout<<getCount(A,n);
    return 0;
}
