#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int A[n];
    for(int i=0;i<n;i++)
        cin>>A[i];
    sort(A,A+n);
    int m, ans=INT_MAX;
    cin>>m;
    for(int beg=0, end=m-1;end<n;beg++,end++)
        ans=min(ans,A[end]-A[beg]);
    cout<<ans;
    return 0;
}
