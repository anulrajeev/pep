import java.util.*;
import java.io.*;

public class Main {

  public static int findSingleElement(int[]arr) {
    int ans=0;
    for(int i=0;i<arr.length;i++)
        ans=ans^arr[i];
    return ans;
  }

  public static void main(String[]args) {
    //input work
    Scanner scn = new Scanner(System.in);
    int n = scn.nextInt();
    int[]arr = new int[n];

    for (int i = 0; i < n; i++) {
      arr[i] = scn.nextInt();
    }

    int ans = findSingleElement(arr);
    System.out.println(ans);
  }
}
