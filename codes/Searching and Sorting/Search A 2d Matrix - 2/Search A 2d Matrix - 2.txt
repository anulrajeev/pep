#include<bits/stdc++.h>
using namespace std;

bool search(vector<vector<int>> &matrix, int val)
{
    int m=matrix.size(), n=matrix[0].size();
    int r=0, c=n-1;
    while(r<m && c>=0)
    {
        if(matrix[r][c]==val)
            return true;
        else if(matrix[r][c]<val)
            r++;
        else
            c--;
    }
    return false;
}

int main()
{
    int m,n, val;
    cin>>m>>n;
    vector<vector<int>> matrix(m,vector<int>(n));
    for(int i=0;i<m;i++)
        for(int j=0;j<n;j++)
            cin>>matrix[i][j];
    cin>>val;
    cout<<(search(matrix,val)?"true":"false");
    
    return 0;
}
