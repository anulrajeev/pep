#include<bits/stdc++.h>
using namespace std;

bool myComp(int a, int b)
{
    string A = to_string(a), B=to_string(b);
    string first=A+B, second=B+A;
    if(first<=second)
        return 1;
    else
        return 0;
}

string largestNumber(int A[], int n)
{
    string res;
    // sort(A,A+n);
    sort(A,A+n,myComp);
    for(int i=n-1;i>=0;i--)
        res+=to_string(A[i]);
    return res;
}

int main()
{
    int n;
    cin>>n;
    int A[n];
    for(int i=0;i<n;i++)
        cin>>A[i];
    cout<<largestNumber(A,n);
    return 0;
}
