#include<bits/stdc++.h>
using namespace std;

bool myComp(int a, int b)
{
    string A=to_string(a), B=to_string(b);
    
    return B+A>A+B;
//    		â—‹ If with A at left and B at right, our required condition is satisfied, then keep them as it is, else reverse them, so that we get the required order.

    
}

string largestNumber(int A[], int n)
{
    string res;
    // sort(A,A+n);
    sort(A,A+n,myComp);
    for(int i=n-1;i>=0;i--)
        res+=to_string(A[i]);
    return res;
}

int main()
{
    int n;
    cin>>n;
    int A[n];
    for(int i=0;i<n;i++)
        cin>>A[i];
    cout<<largestNumber(A,n);
    return 0;
}
