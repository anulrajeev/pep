#include<bits/stdc++.h>
using namespace std;

int largestPerimeter(int A[], int n)
{
    sort(A,A+n);
    for(int i=n-1;i>=2;i--)
        if(A[i]<A[i-1]+A[i-2])
            return A[i]+A[i-1]+A[i-2];
    return 0;
}

int main()
{
    int n;
    cin>>n;
    int A[n];
    for(int i=0;i<n;i++)
        cin>>A[i];
    cout<<largestPerimeter(A,n);
    return 0;
}
