import java.util.*;
import java.io.*;

public class Main {

  public static int firstBadVersion(int n) {
    
    int beg=1, end=n;
    while(beg<=end)
    {
        int mid=beg+(end-beg)/2;
        if(isBadVersion(mid))
        {
            if(mid==0 || !isBadVersion(mid-1))
                return mid;
            end=mid-1;
        }
        else
            beg=mid+1;
    }
    
    return end;
  }

  static int bad = 0;
  public static boolean isBadVersion(int val) {
    if (val >= bad) {
      return true;
    }
    else {
      return false;
    }
  }

  public static void solve(int n, int fbv) {
    bad = fbv;
    System.out.println(firstBadVersion(n));
  }

  public static void main(String[]args) {
    //input work
    Scanner scn = new Scanner(System.in);
    int n = scn.nextInt();
    int fbv = scn.nextInt();

    solve(n, fbv);
  }
}
