#include <iostream>
#include <vector>
#include <queue>
using namespace std;

int findKthLargest(vector<int> const& arr, int k) {

  priority_queue<int, vector<int>, greater<int>> pq;
  int n=arr.size();
  for(int i=0;i<n;i++)
  {
      if(i<k)
        pq.push(arr[i]);
      else
        {
            if(arr[i]>pq.top())
                {
                    pq.push(arr[i]);
                    pq.pop();
                }
        }
    }
  return pq.top();

}

int main()
{
  vector<int>arr;
  int n;
  cin >> n;
  for (int i = 0 ; i < n; i++) {
    int data;
    cin >> data;
    arr.push_back(data);
  }
  int k;
  cin >> k;
  for (int i = k; i > 0; i--)
    cout << findKthLargest(arr, i) << endl;

  return 0;

}
