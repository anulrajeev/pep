    s.insert(arr1[i]);
  }
  cin >> n2;
  int arr2[n2];
  for (int i = 0; i < n2; i++) {
    cin >> arr2[i];

   if(s.count(arr2[i]))
        {

            s.erase(arr2[i]);
        }
            


  }

}
