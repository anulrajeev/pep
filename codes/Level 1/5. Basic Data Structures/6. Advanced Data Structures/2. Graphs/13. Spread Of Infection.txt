#include <bits/stdc++.h>
using namespace std;
typedef vector<int> vi;

int BFS(vi adj[], int V, int src, int t)
{
	bool visited[V]={false};
	queue<int> q;
	q.push(src);
	visited[src]=true;
	int count=1;
	while(q.size() && --t)
	{
		int n=q.size();
		for(int i=0;i<n;i++)
		{
			int u=q.front(); q.pop();
			for(int v : adj[u])
				if(visited[v]==false)
				{
					q.push(v);
					visited[v]=true;
					count++;
				}
		}
	}
	return count;
}

int main()
{
  	int V,k;
  	int u,v,wt,src,t;
  	cin>>V>>k;
  	vi adj[V];
  	for(int i=0;i<k;i++)
  	{
  		cin>>u>>v>>wt;
  		adj[u].push_back(v);
  		adj[v].push_back(u);
  	}
  	cin>>src>>t;
  	cout<<BFS(adj,V,src,t);
  	return 0;
}
