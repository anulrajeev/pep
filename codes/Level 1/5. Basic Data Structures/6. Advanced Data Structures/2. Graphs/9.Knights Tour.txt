void display(vvi &grid)
{
	int n=grid.size();
	for(int i=0;i<n;i++)
	{	for(int j=0;j<n;j++)
			cout<<grid[i][j]<<' ';
		cout<<' ';
	}
	cout<<' ';
}

int dx[]={-2,-1,1,2,2,1,-1,-2};
int dy[]={1,2,2,1,-1,-2,-2,-1};

void knightsTour(vvi &grid, int i, int j, int count)
{
	int n=grid.size();
	if(i<0 || j<0 ||i>=n ||j>=n || grid[i][j])
		return;
	grid[i][j]=count;
	if(count==n*n)
		display(grid);
	else
	   for(int k=0;k<8;k++)
			knightsTour(grid,i+dx[k],j+dy[k],count+1);		
	
	grid[i][j]=0;
}

void _main()
{
  	int n;

  	int r,c;
  	cin>>r>>c;
  	vvi grid(n,vi(n,0));
  	knightsTour(grid,r,c,1);

}
