#include <bits/stdc++.h>
using namespace std;
typedef vector<int> vi;

bool findPath(vi adj[], int V, int src, int dest)
{
	bool visited[V]={false};
	queue<int> q;
	q.push(src);
	visited[src]=true;
	while(q.size())
	{
		int u=q.front(); q.pop();
		for(int v : adj[u])
			if(visited[v]==false)
			{
				if(v==dest)
					return true;
				q.push(v);
			}
	}
	return false;
}

int main()
{
  	int src,dest;
  	int V, edge_count;
  	int u,v,wt;
  	cin>>V>>edge_count;
  	vi adj[V];
  	for(int i=0;i<edge_count;i++)
  	{
  		cin>>u>>v>>wt;
  		adj[u].push_back(v);
   	}
   	cin>>src>>dest;
   	cout<<(findPath(adj,V,src,dest)?"true":"false");
    return 0;
}
