void BFS(vi adj[], int V, int src)
{
	bool visited[V]={false};
	queue<pair<int,string>> q;
	q.push({src,to_string(src)});
	visited[src]=true;
	while(q.size())
	{
		auto p=q.front(); q.pop();
		int u=p.first;
		string psf=p.second;
		cout<<u<<'@'<<psf<<'';
		for(int v : adj[u])
			if(visited[v]==false)
			{
				q.push({v,psf+to_string(v)});
				visited[v]=true;
			}
	}
}

void _main()
{
  	int V,k;
  	int u,v,wt,src;
  	cin>>V>>k;
  	vi adj[V];
  	for(int i=0;i<k;i++)
  	{
  		cin>>u>>v>>wt;
  		adj[u].push_back(v);
  		adj[v].push_back(u);
  	}
  	cin>>src;
  	BFS(adj,V,src);
}
