bool BFS(vi adj[], int V, int src, bool visited[], int colour[])
{
	queue<int> q;
	q.push(src);
	visited[src]=true;
	colour[src]=1;
	while(q.size())
	{
		int u=q.front(); q.pop();
		for(int v : adj[u])
			if(visited[v]==false)
			{
				colour[v]=(colour[u]==1?2:1);
				q.push(v);
				visited[v]=true;
			} 
			else if(colour[v]==colour[u])
				return false;
	}
	return true;
}

bool bipartite(vi adj[], int V)
{
	bool visited[V]={false};
	int colour[V]={0};
	for(int i=0;i<V;i++)
		if(visited[i]==false)
			if(!BFS(adj,V,i,visited,colour))
				return false;
	return true;
}
