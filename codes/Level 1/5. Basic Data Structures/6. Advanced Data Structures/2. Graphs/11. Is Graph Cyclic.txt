bool DFSRec(vi adj[], int V, int u, bool visited[], int parent)
{
	visited[u]=true;
	for(int v : adj[u])
		if(visited[v]==false)
		{
			if(DFSRec(adj,V,v,visited,u))
				return true;
		}
		else if(v!=parent)
			return true;
	return false;
}

bool DFS(vi adj[], int V)
{
	bool visited[V]={false};
	for(int i=0;i<V;i++)
		if(visited[i]==false)
			if(DFSRec(adj,V,i,visited,-1))
				return true;
	return false;
}
