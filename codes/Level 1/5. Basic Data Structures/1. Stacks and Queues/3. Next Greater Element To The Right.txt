#include <bits/stdc++.h>
using namespace std;
void display(vector<int>a) {
  for (int i = 0; i < a.size(); i++)
  {
    cout << a[i] << endl;
  }
}
vector<int> solve(vector<int>A)
{
  stack<int> st;
  vector<int> res(A.size());
  for(int i=A.size()-1;i>=0;i--)
        if(st.size()==0)
        {
            res[i]=-1;
            st.push(A[i]);
        }
        else if(A[i]<st.top())
        {
            res[i]=st.top();
            st.push(A[i]);
        }
        else
        {
            while(st.size() && st.top()<=A[i])
                st.pop();
            if(st.size()==0)
            {
                res[i]=-1;
                st.push(A[i]);
            }
            else
            {
                res[i]=st.top();
                st.push(A[i]);
            }
        }
    return res;
}
int main(int argc, char** argv)
{
  int n;
  cin >> n;
  vector<int>arr(n, 0);
  for (int i = 0; i < n; i++)
  {
    cin >> arr[i];
  }
  vector<int>nge(n, 0);
  nge = solve(arr);
  display(nge);
  return 0;
}
