#include<iostream>
#include<vector>
#include<algorithm>
#include<stack>

using namespace std;
typedef pair<int, int> pii;

int main()
{
    int n;
  	cin>>n;
  	vector<pii> v(n);
  	for(int i=0;i<n;i++)
  		cin>>v[i].first>>v[i].second;
  	sort(v.begin(), v.end());
  	vector<pii> res;
  	res.push_back(v[0]);
  	for(int i=1;i<n;i++)
  	{
  		pii p=v[i];
  		if(p.first<=res.back().second)
  			res.back().second=max(res.back().second, p.second);
  		else
  			res.push_back(p);
  	}
  	for(auto x : res)
  		cout<<x.first<<' '<<x.second<<'
';
    return 0;
}
