#include<iostream>
#include <string>
#include <stack>
using namespace std;



int main() {
  string s;
  cin >> s;
  int n=s.length(), res[n+1]={0}, count=1;
  for(int i=0;i<n;i++)
    if(s[i]=='i')
    {
        int j=i;
        while(j>=0 && res[j]==0)
            res[j--]=count++;
    }
  if(res[n]==0)
  {
      int j=n;
      while(j>=0 && res[j]==0)
        res[j--]=count++;
  }
  for(int i=0;i<=n;i++)
    cout<<res[i];


}
