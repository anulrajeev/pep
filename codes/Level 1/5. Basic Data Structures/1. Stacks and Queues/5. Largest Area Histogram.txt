int largestHistogram(int A[], int n)
{
	int res=0;
	stack<pii> st;
	for(int i=0;i<n;i++)
		if(st.size()==0)
			st.push({A[i], i});
		else if(A[i]>=st.top().first)
			st.push({A[i], i});
		else
		{
			int x,h;
			while(st.size() && A[i]<=st.top().first)
			{
				x = st.top().second, h=st.top().first;
				res=max(res, h*(i-x));
				st.pop();
			}
			st.push({A[i], x});
		}
	while(st.size())
	{
		int x = st.top().second, h=st.top().first;
		res=max(res, h*(n-x));
		st.pop();	
	}
	return res;
}

void _main()
{
  	int n;
  	cin>>n;
  	int A[n];
  	for(int i=0;i<n;i++)
  		cin>>A[i];
  	cout<<largestHistogram(A,n);

}
