#include <bits/stdc++.h>
using namespace std;
typedef vector<string> vs;

string findCelebrity(vs &maze)
{
	stack<int> st;
	int n=maze.size();
	for(int i=0;i<n;i++)
		st.push(i);
	while(st.size()!=1)
	{
		int i=st.top(); st.pop(); 
		int j=st.top(); st.pop();
		if(maze[i][j]=='1')
			st.push(j);
		else
			st.push(i);
	}
	int c=st.top();
	for(int i=0;i<n;i++)
		if(i!=c && maze[i][c]=='0')
			return "none";
	for(int j=0;j<n;j++)
		if(maze[c][j]=='1')
			return "none";
	return to_string(c);
}

int main()
{
  	int n;
  	cin>>n;
  	string s;
  	vs maze(n);
  	for(int i=0;i<n;i++)
  		{
  			cin>>s;
  			maze[i]=s;
  		}	
  	cout<<findCelebrity(maze);
  	return 0;
}
