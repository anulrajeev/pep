#include<iostream>
#include<vector>
#include<string>

using namespace std;

void travel(int i, int j, int dr, int dc, string psf, vector<string> &res)
{
    if(i==dr && j==dc)
        res.push_back(psf);
    else
    {
        int k=1;
        while(j+k<=dc)
            {
                travel(i,j+k,dr,dc,psf+"h"+to_string(k),res);
                k++;
            }
        k=1;
        while(i+k<=dr)
        {
            travel(i+k,j,dr,dc,psf+"v"+to_string(k),res);
                k++;
            }
        k=1;
        while((i+k<=dr)&&(j+k<=dc))
        {
            travel(i+k,j+k,dr,dc,psf+"d"+to_string(k),res);
                    k++;
            }
    }
}



vector<string> get_maze_paths(int sr, int sc, int dr, int dc) {

}

void display(vector<string>& arr) {
  cout << "[";
  for (int i = 0; i < arr.size(); i++) {
    cout << arr[i];
    if (i < arr.size() - 1) cout << ", ";
  }

  cout << "]" << endl;
}
