#include<iostream>
#include<vector>

using namespace std;

// sr - source row
// sc - source column
// dr - destination row
// dc - destination column

void travel(int i, int j, int dr, int dc, string psf, vector<string> &res)
{
    if(i==dr && j==dc)
        res.push_back(psf);
    else
    {
        if(j+1<=dc)
            travel(i,j+1,dr,dc,psf+"h",res);
        if(i+1<=dr)
            travel(i+1,j,dr,dc,psf+"v",res);
    }
}




vector <string> getMazePaths(int sr, int sc, int dr, int dc) {
    vector<string> res;
    travel(sr,sc,dr,dc,"",res);
    return res;
}

void display(vector<string>& arr){
    cout << "[";
    for(int i = 0;i < arr.size();i++){
        cout << arr[i];
        if(i < arr.size() -1) cout << ", ";
    }
    
    cout << "]"<<endl;
}


int main() {
    int n,m; cin >> n >> m;
    vector<string> ans = getMazePaths(0,0,n-1,m-1);
    display(ans);

    return 0;
}
