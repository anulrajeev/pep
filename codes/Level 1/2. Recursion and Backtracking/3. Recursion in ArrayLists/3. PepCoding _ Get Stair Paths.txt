#include<iostream>
#include<vector>
#include<string>
using namespace std;

void climb(int i, int n, string steps, vector<string> &res)
{
    if(i==n)
        res.push_back(steps);
    else
        {
            if(i+1<=n)
                climb(i+1,n,steps+"1",res);
            if(i+2<=n)
                climb(i+2,n,steps+"2",res);
            if(i+3<=n)
                climb(i+3,n,steps+"3",res);
        }
    
        
}



vector<string> get_stair_paths(int n) {
  vector<string> res;
  climb(0,n,"",res);
  return res;
}

int main() {
  int n;
  cin >> n;
  vector<string> ans = get_stair_paths(n);
  int cnt = 0;

  cout << '[';
  for (string str : ans) {
    if (cnt != ans.size() - 1)
      cout << str << ", ";
    else
      cout << str;
    cnt++;
  }
  cout << ']';
}
