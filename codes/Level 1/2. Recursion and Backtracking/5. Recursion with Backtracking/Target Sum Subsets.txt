void subsetSum(int A[], int n, int i, int sum, string psf)
{
	if(i==n)
	{
		if(sum==0)
			cout<<psf<<".
";
		return;
	}
	else
	{
		if(A[i]<=sum)
			subsetSum(A,n,i+1,sum-A[i],psf+to_string(A[i])+", ");

		subsetSum(A,n,i+1,sum,psf);
	}

}

void _main()
{
  	int n, target;
  	cin>>n;
  	int A[n];
  	for(int i=0;i<n;i++)
  		cin>>A[i];
  	cin>>target;

  	subsetSum(A,n,0,target,"");

}
