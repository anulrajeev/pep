#include <bits/stdc++.h>
using namespace std;

void countSortr(int A[], int n, int e, int mod)
{
    int size=(e==1?10000:100);
	int count[size]={0};
	for(int i=0;i<n;i++)
		count[(A[i]/e)%mod]++;
	for(int i=1;i<size;i++)
		count[i]+=count[i-1];

	int output[n];
	for(int i=n-1;i>=0;i--)
	{
		output[count[(A[i]/e)%mod]-1]=A[i];
		count[(A[i]/e)%mod]--;
	}

	for(int i=0;i<n;i++)
		A[i]=output[i];

}


void radixSort(int A[], int n)
{
	countSortr(A,n,pow(10,6),100);
	countSortr(A,n,pow(10,4),100);
	countSortr(A,n,1,10000);
}

int main()
{
	int n;
	cin>>n;
	int A[n];
	for(int i=0;i<n;i++)
		cin>>A[i];

	radixSort(A,n);
  	
  	for(int i=0;i<n;i++)
  	{
  	    if(A[i]<pow(10,7))
  	        cout<<0;
  		cout<<A[i]<<'
';
  	}
  	return 0;
}
