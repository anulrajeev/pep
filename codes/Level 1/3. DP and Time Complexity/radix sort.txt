#include <bits/stdc++.h>
using namespace std;

void countSortr(int A[], int n, int e)
{
	int count[10]={0};
	for(int i=0;i<n;i++)
		count[(A[i]/e)%10]++;
	for(int i=1;i<10;i++)
		count[i]+=count[i-1];

	int output[n];
	for(int i=n-1;i>=0;i--)
	{
		output[count[(A[i]/e)%10]-1]=A[i];
		count[(A[i]/e)%10]--;
	}

	for(int i=0;i<n;i++)
		A[i]=output[i];
	cout<<"After sorting on "<<e<<" place -> ";
	for(int i=0;i<n;i++)
	    cout<<A[i]<<' ';
	cout<<'
';
}


void radixSort(int A[], int n)
{
	int d=log10(*max_element(A,A+n))+1;
	for(int i=0;i<d;i++)
		countSortr(A,n,pow(10,i));
}

int main()
{
	int n;
	cin>>n;
	int A[n];
	for(int i=0;i<n;i++)
		cin>>A[i];

	radixSort(A,n);
  	
  	for(int i=0;i<n;i++)
  			cout<<A[i]<<' ';
  	return 0;
}
