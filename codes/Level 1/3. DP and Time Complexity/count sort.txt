#include <bits/stdc++.h>
using namespace std;

void countSort(int A[], int n)
{
	int N=*max_element(A,A+n)+1, d=*min_element(A,A+n);
	int count[N-d]={0};
	for(int i=0;i<n;i++)
		count[A[i]-d]++;
	for(int i=1;i<N-d;i++)
		count[i]+=count[i-1];

	int output[n];
	for(int i=n-1;i>=0;i--)
	{
		output[count[A[i]-d]-1]=A[i];
		count[A[i]-d]--;
	}

	for(int i=0;i<n;i++)
		A[i]=output[i];
}

int main()
{
	int n;
	cin>>n;
	int A[n];
	for(int i=0;i<n;i++)
		cin>>A[i];
	countSort(A,n);
  	
  	for(int i=0;i<n;i++)
  			cout<<A[i]<<'
';
  	return 0;
}
