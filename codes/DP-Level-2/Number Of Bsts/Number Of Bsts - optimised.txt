ll C(int n, int r)
{
	ll res=1;
	for(int i=0;i<r;i++)
	{
		res=res*(n-i);
		res=res/(i+1);
	}
	return res;
}

ll countBST(int n)
{
	int dp[n+1]={1,1};
	for(int i=2;i<=n;i++)
	    for(int j=0;j<i;j++)
	        dp[i]+=dp[j]*dp[i-j-1];
	 return dp[n];
}
