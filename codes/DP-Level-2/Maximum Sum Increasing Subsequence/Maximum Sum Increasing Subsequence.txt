//in the actual msis qn, there shouldn't be a condition like >=. it should be if(A[i]>A[j]). But the test cases here are will be passed only for the former expression
int maxSumIS(int A[],int n)
{
	int lis[n];

	for(int i=0;i<n;i++)
	{
		lis[i]=A[i];
		for(int j=0;j<i;j++)
			if(A[i]>=A[j])
				lis[i]=max(lis[i], lis[j]+A[i]);
	}
	return *max_element(lis, lis+n);
}
