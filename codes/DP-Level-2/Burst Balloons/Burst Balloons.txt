int maxCoins(int start, int end, vi &v, vvi &dp)
{
	if(start>end)
		return 0;
	if(dp[start][end]!=-1)
		return dp[start][end];

	int ans = INT_MIN;
	for(int lastBust = start; lastBust<=end; lastBust++)
	{
		int score = v[start-1]*v[lastBust]*v[end+1]+maxCoins(start,lastBust-1,v,dp)+maxCoins(lastBust+1,end,v,dp);
		ans = max(ans, score);
	}
	return dp[start][end]=ans;
}


int burstBalloons(int A[], int n)
{
	vi v;
	v.push_back(1);
	for(int i=0;i<n;i++)
		v.push_back(A[i]);
	v.push_back(1);
	vvi dp(n+1,vi(n+1,-1));
	return maxCoins(1,n,v,dp);
}

		cin>>A[i];  	
	cout<<burstBalloons(A,n);
}
