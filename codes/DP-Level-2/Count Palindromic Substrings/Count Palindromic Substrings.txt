int f(string s)
{

	int n=s.length();
	int dp[n][n], count=0;;
	for(int gap=0;gap<n;gap++)
		for(int i=0;i+gap<n;i++)
		{
			int j=i+gap;
			if(i==j)
				dp[i][j]=1;
			else if(i+1==j)
				dp[i][j]=(s[i]==s[j]);
			else
				if(s[i]==s[j])
					dp[i][j]=dp[i+1][j-1];
				else
					dp[i][j]=0;
			count+=dp[i][j];
		}
	return count;
}
