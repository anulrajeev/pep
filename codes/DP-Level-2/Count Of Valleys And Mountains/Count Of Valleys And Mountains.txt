#include<iostream>
using namespace std;

int f(int n)
{
    if(n==0 || n==1)
        return n;
    int dp[n+1]={1,1};
    for(int i=2;i<=n;i++)
    {
        dp[i]=0;
        for(int j=i-1;j>=0;j--)
            dp[i]+=(dp[j]*dp[i-1-j]);
        
    }
    return dp[n];
}

int main()
{
    int n;
    cin>>n;
    cout<<f(n);
    return 0;
}
