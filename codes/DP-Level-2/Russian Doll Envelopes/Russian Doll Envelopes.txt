int max_non_overlappingBridges(vector<pii> &v, int n)
{
	sort(v.begin(), v.end());
	
	int lis[n];

	for(int i=0;i<n;i++)
	{
		lis[i]=1;
		for(int j=0;j<i;j++)
			if(v[i].second>v[j].second)
				lis[i]=max(lis[i], lis[j]+1);
	}

	return *max_element(lis, lis+n);
}
