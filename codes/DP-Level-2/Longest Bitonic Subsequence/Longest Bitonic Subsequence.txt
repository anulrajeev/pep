int longestBitonic(int A[], int n)
{
	int lis[n];

	for(int i=0;i<n;i++)
	{
		lis[i]=1;
		for(int j=0;j<i;j++)
			if(A[i]>A[j])
				lis[i]=max(lis[i], lis[j]+1);
	}

	int lds[n];

	for(int i=n-1;i>=0;i--)
	{
		lds[i]=1;
		for(int j=i+1;j<n;j++)
			if(A[i]>A[j])
				lds[i]=max(lds[i], lds[j]+1);

	}

	int res=0;
	for(int i=0;i<n;i++)
		res=max(res, lis[i]+lds[i]);
	return res-1;

}

void _main()

	cin>>n;
	int A[n];
  	for(int i=0;i<n;i++)
  		cin>>A[i];
  	cout<<longestBitonic(A,n);
}
