int maxScoreTriangulation(int A[], int n)
{
	vvi dp(n, vi(n, 0));
	for(int gap=0;gap<=2;gap++)
		for(int i=0;i+gap<n;i++)
		{
			int j=i+gap;
			if(gap==0)
				dp[i][j]=0;
			else if(gap==1)
				dp[i][j]=0;
			else if(gap==2)
				dp[i][j]=A[i]*A[i+1]*A[i+2];
		}

	// for(int i=0;i<n;i++)
	// {	for(int j=0;j<n;j++)
	// 		cout<<dp[i][j]<<'	';
	// 	cout<<'
';
	// }

	for(int gap=3;gap<n;gap++)
		for(int i=0;i+gap<n;i++)
		{
			int j=i+gap;
			int res=INT_MAX;
			for(int k=1;i+k<j;k++)
			{
				int tria  = A[i]*A[i+k]*A[j];
				int leftp = dp[i][i+k];
				int rightp= dp[i+k][j];
				res=min(res, tria+leftp+rightp);
			}

			dp[i][j]=res;
		}
	return dp[0][n-1];
}
