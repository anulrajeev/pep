#include<iostream>
using namespace std;

long long f(int N)
{
    if(N==0)
        return 0;
    if(N==2)
        return 1;
    long long dp[N+1]={1,0,1};
    for(int i=4;i<=N;i+=2)
    {
        dp[i]=0;
        for(int j=0;j<=i-2;j+=2)
            dp[i]+=(dp[j]*dp[i-2-j]);
    }
    

    return dp[N];
    
}

int main()
{
    int n;
    cin>>n;
    cout<<f(2*n);
}
